import itertools
import numpy as np




[docs]def polyfit2d(x, y, z, order=3):
    '''
    fit unstructured data 
    '''
    ncols = (order + 1)**2
    G = np.zeros((x.size, ncols))
    ij = itertools.product(range(order+1), range(order+1))
    for k, (i,j) in enumerate(ij):
        G[:,k] = x**i * y**j
    m = np.linalg.lstsq(G, z)[0]
    return m




[docs]def polyval2d(x, y, m, dtype=None):
    order = int(np.sqrt(len(m))) - 1
    ij = itertools.product(range(order+1), range(order+1))
    z = np.zeros_like(x, dtype=dtype)
    for a, (i,j) in zip(m, ij):
        z += a * x**i * y**j
    return z




[docs]def polyfit2dGrid(arr, mask, order=3, copy=False):
    '''
    replace all masked values with polynomal fitted ones
    '''
    x,y = np.where(~mask)
    z = arr[~mask]
    p = polyfit2d(x,y,z,order)
    xx,yy = np.where(mask)
    if copy:
        arr = arr.copy()
    arr[mask] = polyval2d(xx, yy, p, dtype=arr.dtype)
    return arr
